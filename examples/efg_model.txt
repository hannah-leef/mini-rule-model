# molecule types
Molecule EGFR (l, y~U~P)
Molecule EGF (r)

# initial amount (uM)
INIT EGF 1.0
INIT EGFR 0.1

# ligand binding
Rule bind: EGF(r) + EGFR(l) -> EGF(r).EGFR(l)   kf=1e6 kr=1e-2

# simplified dimerization
Rule dimer: EGFR(l).EGFR(l) -> EGFR(l).EGFR(l)   kf=1e5

# trans-phosphorylation
Rule phos: EGFR(y~U).EGFR(y~U) -> EGFR(y~P).EGFR(y~U)  kf=0.1

# observable example
Observable EGFR_P EGFR(y~P)
